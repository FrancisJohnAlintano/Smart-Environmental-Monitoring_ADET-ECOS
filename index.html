def send_environmental_data(device_id, data_type, value):
    try:
        txn = contract.functions.storeData(
            device_id, data_type, str(value)
        ).transact({
            'from': owner_address,
            'gas': 500000
        })
        receipt = web3.eth.wait_for_transaction_receipt(txn)
        print(f"✅ Data Stored: {data_type} = {value}")
        return True
    except Exception as e:
        print(f"❌ Error: {str(e)}")
        return False
